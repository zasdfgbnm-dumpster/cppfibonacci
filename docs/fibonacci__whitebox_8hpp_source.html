61  static size_t _data_structure_consistency_test(ss_t node, ss_t parent, ss_t head) {
62  // if node is null, head must also be null
63  if(!node)
64  if(head) throw unexpected_nullptr;
65  // if sibling test done
66  if (node==head)
67  return 0;
68  // if this is the beginning of sibling test
69  if (head==nullptr) head = node;
70  // test min-tree property
71  if(parent)
72  if(Compare()(node->data->key,parent->data->key)) throw min_tree_property_violation;
73  // test parent and sibling pointers
74  if(node->parent.lock()!=parent) throw wrong_parent_pointer;
75  if(node->left_sibling.expired())
76  throw null_left_sibling_pointer;
77  else if(node->left_sibling.lock()->right_sibling!=node)
78  throw doubly_linked_list_property_violation;
79  if(!node->right_sibling)
80  throw null_right_sibling_pointer;
81  else if(node->right_sibling->left_sibling.lock()!=node)
82  throw doubly_linked_list_property_violation;
83  // test structure and data pointers
84  if(!node->data)
85  throw null_data_pointer;

99  static bool _is_binomial(ss_t root) {
100  if(!root) return true;
101  size_t degree = root->degree;
102  if(degree==0){
103  if(root->child) throw bad_degree;
104  else return true;
105  }
106  bool children_degrees[degree];
107  for(bool &i:children_degrees)
108  i = false;
109  ss_t p=root->child;
110  do {
111  if(!p) throw unexpected_nullptr;
112  bool subtreetest = _is_binomial(p);
113  if(!subtreetest) return false;
114  if(p->degree<0||p->degree>=degree) return false;
115  if(children_degrees[p->degree]) return false;
116  children_degrees[p->degree] = true;
117  p=p->right_sibling;
118  } while(p!=root->child);
119  for(bool i:children_degrees)
120  if(!i) return false;
121  return true;
122  }

154  // if pointers are null, return

156  bool anynull = false;
157  for(ss_t &i:nodes) {

159  anynull = anynull || (i==nullptr);

168  for(ss_t &p1:ps){
169  for(ss_t &p2:ps){

171  if(p1->data->data!=p2->data->data) return false;
