164  static void meld(ssp &target, ssp node, bool update_parent, bool find_min, bool set_min, bool reset_childcut, ssp parent=nullptr) {
165  if(!node) return;
166  if(target) parent = target->parent.lock();
167  // update parent and find the min element
168  // if neither updating the parent nor the finding min, nor reseting childcut
169  // is needed, avoid this O(m) loop
170  if ( update_parent || find_min || reset_childcut ) {
171  ssp oldhead = node;
172  ssp p = oldhead;
173  do {
174  if(update_parent) p->parent = parent;
175  if(reset_childcut) p->childcut = false;
176  if(find_min && Compare()(p->data->key,node->data->key))
177  node = p;
178  p=p->right_sibling;
179  } while(p!=oldhead);
180  }
181  // merge sibling lists
182  if(target) {
183  target->right_sibling.swap(node->right_sibling);
184  target->right_sibling->left_sibling.swap(node->right_sibling->left_sibling);
185  if( set_min && Compare()(node->data->key,target->data->key) )
186  target = node;
187  } else {
188  target = node;
189  }
190  }

440  node remove() {
441  if(_size==0) throw "no element to remove";
442  ssp oldmin = min;
443  if(_size==1) {
444  _size = 0;
445  min = nullptr;
446  return node(oldmin);

448 
449  // merge trees of same degrees
450  std::vector<ssp> trees(max_degree()+1);
451  if(min->child)
452  meld(min,min->child,false,false,false,false);
453  while(min->right_sibling!=min) {
454  ssp q = min->right_sibling;
455  remove_tree(q);
456  while(trees[q->degree]) {
457  bool q_is_smaller = Compare()(q->data->key,trees[q->degree]->data->key);
458  ssp smaller = q_is_smaller?q:trees[q->degree];
459  ssp larger = q_is_smaller?trees[q->degree]:q;
460  trees[q->degree] = nullptr;
461  meld(smaller->child,larger,true,false,false,true,smaller);
462  smaller->degree++;
463  q = smaller;
464  }
465  trees[q->degree] = q;
466  }

469  // cut the loop so that the resource for deleted structure node can be
470  // cleaned up by std::shared_ptr.
471  min->right_sibling = nullptr;
472  min = nullptr;
473  for(ssp p:trees) {
474  if(!p) continue;
